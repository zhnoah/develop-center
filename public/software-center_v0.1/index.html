  • MongoDB
  • // 创建用于存放数据及日志的文件夹
mkdir e:mongodb\data\db
mkdir e:mongodb\data\log

// 创建配置文件
在 mongodb 文件夹下创建文件 mongod.cfg，内容：
systemLog:
  destination: file
  path: e:\mongodb\data\log\mongod.log
storage:
  dbPath: e:\mongodb\data\db

// 安装 MongoDB 服务
"{Path}\bin\mongod.exe" --config "e:\mongodb\mongod.cfg" --install

// 启动/停止
net start MongoDB
net stop MongoDB

// 移除服务
"{Path}\bin\mongod.exe" --remove
